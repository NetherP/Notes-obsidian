```bash
cp source target

mv source target

rm target
```
## cp - copy file
options
- -r=recursive
- -a=retain date and permission
- -i=prompt for overwrite

## mv - move file
#### options 
- -i=prompt for overwrite
- -b= make backup of target if its already exist

## rm - remove file
#### options
- -i=prompt for each
- -r=recursive
- -f=remove without prompting
